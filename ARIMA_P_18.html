# Create DataFrame
df = pd.DataFrame(data)
df.set_index('YEAR', inplace=True)

# Transform PDI to log scale
df['log_PDI'] = np.log(df['PDI'])

# Plot the log PDI
plt.figure(figsize=(12,6))
plt.plot(df['log_PDI'])
plt.title('Log of Personal Disposable Income (1970-1991)')
plt.xlabel('Year')
plt.ylabel('Log PDI')
plt.grid(True)
plt.show()

# Function to test stationarity
def test_stationarity(timeseries):
    result = adfuller(timeseries)
    print('ADF Statistic:', result[0])
    print('p-value:', result[1])
    print('Critical Values:', result[4])
    if result[1] <= 0.05:
        print("Series is stationary")
    else:
        print("Series is not stationary")

# Test stationarity of log PDI
print("Stationarity test for log PDI:")
test_stationarity(df['log_PDI'])

# Difference the series and test stationarity
df['log_PDI_diff'] = df['log_PDI'].diff().dropna()
print("\nStationarity test for differenced log PDI:")
test_stationarity(df['log_PDI_diff'].dropna())

# Plot ACF and PACF
plt.figure(figsize=(12,4))
plt.subplot(121)
plot_acf(df['log_PDI_diff'].dropna(), ax=plt.gca())
plt.subplot(122)
plot_pacf(df['log_PDI_diff'].dropna(), ax=plt.gca())
plt.show()

# Fit ARIMA model (1,1,1) based on ACF/PACF analysis
model = ARIMA(df['log_PDI'], order=(1,1,1))
results = model.fit()

# Print model summary
print(results.summary())

# Plot diagnostics
results.plot_diagnostics(figsize=(12,8))
plt.show()

# Forecast next 4 quarters
forecast = results.forecast(steps=4)
forecast_years = [1992.1, 1992.2, 1992.3, 1992.4]

# Convert forecast back to original scale
forecast_original_scale = np.exp(forecast)

# Plot forecast

plt.ylabel('PDI (Billions of 1987 $)')
plt.legend()
plt.grid(True)
plt.show()

# Print forecast values
print("\nForecast for 1992 (in billions of 1987 $):")
for year, value in zip(forecast_years, forecast_original_scale):
    print(f"Q{year}: {value:.1f}")
