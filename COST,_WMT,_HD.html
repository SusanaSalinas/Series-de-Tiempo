import yfinance as yf
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from statsmodels.tsa.stattools import adfuller

# Fetch historical stock data for COST from yfinance
start_date = "2019-11-29"
end_date = "2025-02-25"
ticker = "COST"
cost_data = yf.download(ticker, start=start_date, end=end_date)

# Perform a random walk test (Dickey-Fuller test)
def adf_test(timeseries):
    result = adfuller(timeseries, autolag='AIC')
    return result[1]  # Return the p-value

# Calculate moving averages
cost_data['MA9'] = cost_data['Close'].rolling(window=9).mean()
cost_data['MA30'] = cost_data['Close'].rolling(window=30).mean()

# Visualize the stock prices along with moving averages
plt.figure(figsize=(14, 7))
plt.plot(cost_data['Close'], label='Closing Price')
plt.plot(cost_data['MA9'], label='9-Day Moving Average')
plt.plot(cost_data['MA30'], label='30-Day Moving Average')
plt.title('COST Stock Price with Moving Averages')
plt.xlabel('Date')
plt.ylabel('Price')
plt.legend()
plt.grid(True)
plt.show()

# Random Walk and Unit Root Test
close_prices = cost_data['Close'].dropna()
p_value = adf_test(close_prices)

print(f'ADF Test p-value: {p_value}')
if p_value < 0.05:
    print("The time series is stationary (reject the null hypothesis of a unit root).")
else:
    print("The time series is non-stationary (fail to reject the null hypothesis of a unit root).")
